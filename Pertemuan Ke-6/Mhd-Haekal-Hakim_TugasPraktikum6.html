Uji Box-cox

# Bikin Index
index <- seq_along(dt_rg)
# Buat Data Supaya Positif
shift_val <- abs(min(dt_rg)) + 1
dt_rg_shift <- dt_rg + shift_val
# Regresi Linear untuk Box-Cox
lm_bc <- lm(dt_rg_shift ~ index)
# Uji Box-Cox
bc <- boxcox(lm_bc, lambda = seq(0, 5, by = 0.001))

# Cari Lambda Optimum
lambda_opt <- bc$x[which.max(bc$y)]
ci_lambda <- bc$x[bc$y > max(bc$y) - 0.5 * qchisq(0.95, 1)]

cat("Lambda Optimum:", lambda_opt, "\n")
## Lambda Optimum: 2.788
cat("95% CI Lambda:", min(ci_lambda), "-", max(ci_lambda), "\n")
## 95% CI Lambda: 1.402 - 4.206

Transformasi

# --- Transformasi Box-Cox ---
# Pastikan semua nilai positif
shift_val <- abs(min(dt_rg)) + 1
dt_shift <- dt_rg + shift_val

# Regresi linear untuk uji Box-Cox
lm_bc <- lm(dt_shift ~ seq_along(dt_shift))

# Uji Box-Cox dengan rentang lambda yang cukup luas
bc <- boxcox(lm_bc, lambda = seq(-1, 2, 0.001))

# Ambil lambda optimum
lambda_opt <- bc$x[which.max(bc$y)]
cat("Lambda optimum Box-Cox:", lambda_opt, "\n")
## Lambda optimum Box-Cox: 2
# Transformasi Box-Cox
dt_bc <- BoxCox(dt_shift, lambda = lambda_opt)

Selain dengan eksplorasi, asumsi tersebut dapat diuji menggunakan uji formal. Pada tahapan ini uji formal yang digunakan untuk normalitas adalah uji Kolmogorov-Smirnov (KS). Hipotesis pada uji KS adalah sebagai berikut.

\(H_1\): Sisaan tidak menyebar normal
